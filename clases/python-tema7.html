Proyecto Final — Gestor de Red Completo

Integra todo lo aprendido: script completo que conecta a N dispositivos, hace backup, verifica configuración, parsea outputs y genera reporte HTML profesional. El resultado de los Temas 1-6 en un sistema de producción.

🐍 Python 3.x 🎯 CCNA Domain 6.4 ⏱ ~120 min práctica 📦 Netmiko + NAPALM + TextFSM 🌐 Reporte HTML Dinámico

El Proyecto Integrador

Este es el momento de unir todo: Python esencial, Netmiko para conexiones, TextFSM para parsear, NAPALM para configuraciones estandarizadas, y generación de reportes. El resultado es un sistema de gestión de red completo que cualquier empresa usaría en producción.

🔄

Conecta

SSH a routers, switches y firewalls usando Netmiko con manejo de errores robusto y timeouts configurables.

💾

Respalda

Backup automático de configuraciones con timestamp, almacenamiento organizado y validación de integridad.

🔍

Verifica

Parseo con TextFSM de interfaces, rutas y vecinos. Comparación contra baseline de seguridad.

📊

Reporta

Dashboard HTML profesional con estado de salud, gráficos de cumplimiento y detalles por dispositivo.

Cargador de Inventario

inventory_loader.py
import yaml
from dataclasses import dataclass
from typing import List, Optional

@dataclass
class NetworkDevice:
    hostname: str
    host: str
    role: str
    site: str
    device_type: str
    username: str
    password: str
    status: str = "unknown"  # Para tracking

class InventoryManager:
    def __init__(self, inventory_file):
        self.inventory_file = inventory_file
        self.devices: List[NetworkDevice] = []
        self._load()
    
    def _load(self):
        """Carga YAML y crea objetos Device"""
        with open(self.inventory_file) as f:
            data = yaml.safe_load(f)
        
        defaults = data.get('defaults', {})
        
        for group, devices in data.items():
            if group == 'defaults':
                continue
            
            for dev in devices:
                # Merge con defaults
                device_config = {**defaults, **dev}
                
                self.devices.append(NetworkDevice(
                    hostname=device_config['hostname'],
                    host=device_config['host'],
                    role=device_config.get('role', 'unknown'),
                    site=device_config.get('site', 'default'),
                    device_type=device_config['device_type'],
                    username=device_config['username'],
                    password=device_config['password']
                ))
    
    def get_by_role(self, role: str) -> List[NetworkDevice]:
        return [d for d in self.devices if d.role == role]
    
    def get_all(self) -> List[NetworkDevice]:
        return self.devices

from netmiko import ConnectHandler, NetMikoTimeoutException, NetMikoAuthenticationException
from napalm import get_network_driver
import logging

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

class ConnectionManager:
    """Gestiona conexiones SSH con reintentos y logging"""
    
    def __init__(self, device):
        self.device = device
        self.connection = None
        self.connected = False
    
    def connect_netmiko(self, retries=3):
        """Conexión SSH con reintentos"""
        device_params = {
            "device_type": self.device.device_type,
            "host": self.device.host,
            "username": self.device.username,
            "password": self.device.password,
            "timeout": 30,
            "secret": self.device.password  # Para enable
        }
        
        for attempt in range(retries):
            try:
                logger.info(f"Conectando a {self.device.hostname} (intento {attempt+1})")
                self.connection = ConnectHandler(**device_params)
                self.connection.enable()  # Entrar a privileged mode
                self.connected = True
                self.device.status = "connected"
                return True
                
            except NetMikoTimeoutException:
                logger.error(f"Timeout en {self.device.hostname}")
            except NetMikoAuthenticationException:
                logger.error(f"Auth failed en {self.device.hostname}")
                break  # No reintentar auth
            except Exception as e:
                logger.error(f"Error en {self.device.hostname}: {e}")
        
        self.device.status = "failed"
        return False
    
    def disconnect(self):
        if self.connection:
            self.connection.disconnect()
            self.connected = False
    
    def send_command(self, command, use_textfsm=False):
        """Envía comando y retorna output estructurado si se pide"""
        if not self.connected:
            raise Exception("No conectado")
        
        try:
            output = self.connection.send_command(
                command, 
                use_textfsm=use_textfsm,
                delay_factor=2
            )
            return output
        except Exception as e:
            logger.error(f"Error enviando comando: {e}")
            return None
    
    def get_config(self, config_type="running"):
        """Obtiene configuración del dispositivo"""
        if config_type == "running":
            return self.connection.send_command("show running-config")
        return None

Ejecución Paralela

Parallel executor para múltiples dispositivos
from concurrent.futures import ThreadPoolExecutor, as_completed

def process_device(device, commands):
    """Procesa un dispositivo: conecta, colecta, desconecta"""
    results = {
        "hostname": device.hostname,
        "host": device.host,
        "commands": {},
        "config": None,
        "errors": []
    }
    
    conn = ConnectionManager(device)
    
    if not conn.connect_netmiko():
        results["errors"].append("Connection failed")
        return results
    
    try:
        # Backup
        results["config"] = conn.get_config()
        
        # Comandos de verificación
        for cmd in commands:
            output = conn.send_command(cmd, use_textfsm=True)
            results["commands"][cmd] = output
            
    except Exception as e:
        results["errors"].append(str(e))
    finally:
        conn.disconnect()
    
    return results

# Ejecutar en paralelo
def run_parallel_audit(devices, commands, max_workers=5):
    all_results = []
    
    with ThreadPoolExecutor(max_workers=max_workers) as executor:
        future_to_device = {
            executor.submit(process_device, dev, commands): dev 
            for dev in devices
        }
        
        for future in as_completed(future_to_device):
            result = future.result()
            all_results.append(result)
    
    return all_results

Sistema de Backups Automatizados

Backup diario con organización por fecha, validación de integridad (hash), y retención automática.

backup_manager.py
import os
import hashlib
from datetime import datetime
from pathlib import Path

class BackupManager:
    def __init__(self, base_path="backups"):
        self.base_path = Path(base_path)
        self.today = datetime.now().strftime("%Y-%m-%d")
        self.backup_dir = self.base_path / self.today
        self.backup_dir.mkdir(parents=True, exist_ok=True)
    
    def save_config(self, hostname, config):
        """Guarda configuración con metadata"""
        timestamp = datetime.now().strftime("%H%M%S")
        filename = f"{hostname}_{timestamp}.cfg"
        filepath = self.backup_dir / filename
        
        # Guardar
        with open(filepath, 'w') as f:
            f.write(config)
        
        # Calcular hash para integridad
        file_hash = self._calculate_hash(filepath)
        
        # Metadata
        meta = {
            "hostname": hostname,
            "timestamp": datetime.now().isoformat(),
            "file": str(filepath),
            "hash_md5": file_hash,
            "size_bytes": os.path.getsize(filepath)
        }
        
        # Guardar metadata
        meta_file = self.backup_dir / f"{hostname}_{timestamp}.json"
        import json
        with open(meta_file, 'w') as f:
            json.dump(meta, f, indent=2)
        
        return meta
    
    def _calculate_hash(self, filepath):
        """MD5 para verificar integridad"""
        hash_md5 = hashlib.md5()
        with open(filepath, "rb") as f:
            for chunk in iter(lambda: f.read(4096), b""):
                hash_md5.update(chunk)
        return hash_md5.hexdigest()
    
    def cleanup_old_backups(self, days=30):
        """Elimina backups antiguos"""
        from datetime import timedelta
        
        cutoff = datetime.now() - timedelta(days=days)
        
        for item in self.base_path.iterdir():
            if item.is_dir():
                try:
                    dir_date = datetime.strptime(item.name, "%Y-%m-%d")
                    if dir_date < cutoff:
                        import shutil
                        shutil.rmtree(item)
                        print(f"Eliminado backup antiguo: {item.name}")
                except ValueError:
                    pass  # No es directorio de fecha

import difflib

def compare_to_baseline(running_config, baseline_file):
    """Compara config actual vs estándar"""
    with open(baseline_file) as f:
        baseline = f.readlines()
    
    running = running_config.splitlines(keepends=True)
    
    diff = difflib.unified_diff(
        baseline, running,
        fromfile='baseline',
        tofile='running',
        lineterm=''
    )
    
    return ''.join(diff)

from ntc_templates.parse import parse_output

class NetworkAnalyzer:
    """Analiza datos recolectados y detecta problemas"""
    
    def analyze_interfaces(self, raw_output, platform="cisco_ios"):
        """Analiza estado de interfaces"""
        try:
            parsed = parse_output(
                platform=platform,
                command="show ip interface brief",
                data=raw_output
            )
            
            stats = {
                "total": len(parsed),
                "up_up": 0,
                "down_down": 0,
                "admin_down": 0,
                "issues": []
            }
            
            for intf in parsed:
                status = intf.get('status', '')
                protocol = intf.get('protocol', '')
                
                if status == 'up' and protocol == 'up':
                    stats['up_up'] += 1
                elif status == 'administratively down':
                    stats['admin_down'] += 1
                elif status == 'down':
                    stats['down_down'] += 1
                    # Detectar problemas
                    if intf.get('ip_address') != 'unassigned':
                        stats['issues'].append(
                            f"{intf['interface']}: IP asignada pero interfaz down"
                        )
            
            return stats
            
        except Exception as e:
            return {"error": str(e)}
    
    def check_compliance(self, config_text):
        """Verifica políticas de seguridad básicas"""
        checks = {
            "ssh_v2": "ip ssh version 2" in config_text,
            "no_telnet": "transport input telnet" not in config_text,
            "password_encryption": "service password-encryption" in config_text,
            "ntp_configured": "ntp server" in config_text,
            "logging_buffered": "logging buffered" in config_text
        }
        
        score = sum(1 for v in checks.values() if v) / len(checks) * 100
        
        return {
            "checks": checks,
            "score": round(score, 1),
            "compliant": score == 100
        }
    
    def extract_version_info(self, raw_output):
        """Extrae información de versión"""
        parsed = parse_output(
            platform="cisco_ios",
            command="show version",
            data=raw_output
        )
        return parsed[0] if parsed else {}

from jinja2 import Environment, FileSystemLoader
from datetime import datetime
import json

class ReportGenerator:
    def __init__(self, template_dir="templates"):
        self.env = Environment(loader=FileSystemLoader(template_dir))
        self.template = self.env.get_template("report.html")
    
    def generate_html(self, data, output_file):
        """Genera reporte HTML"""
        report_data = {
            "date": datetime.now().strftime("%Y-%m-%d %H:%M:%S"),
            "summary": self._calculate_summary(data),
            "devices": data
        }
        
        html_output = self.template.render(**report_data)
        
        with open(output_file, 'w', encoding='utf-8') as f:
            f.write(html_output)
        
        return output_file
    
    def generate_json(self, data, output_file):
        """Exporta datos en JSON para APIs"""
        with open(output_file, 'w') as f:
            json.dump(data, f, indent=2, default=str)
    
    def _calculate_summary(self, data):
        """Calcula métricas agregadas"""
        total = len(data)
        online = sum(1 for d in data if d.get('status') == 'online')
        avg_compliance = sum(d.get('compliance_score', 0) for d in data) / total if total > 0 else 0
        total_issues = sum(len(d.get('issues', [])) for d in data)
        
        return {
            "total_devices": total,
            "online": online,
            "compliant": round(avg_compliance, 1),
            "backups": sum(1 for d in data if d.get('backup_taken')),
            "issues": total_issues
        }

                results.append(processed)
                
            except Exception as e:
                logger.error(f"Error procesando {device.hostname}: {e}")
                results.append({
                    "hostname": device.hostname,
                    "ip": device.host,
                    "status": "error",
                    "error": str(e)
                })
    
    # 4. Generar reportes
    timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
    
    reporter = ReportGenerator()
    
    # HTML Report
    html_file = f"{REPORT_DIR}/report_{timestamp}.html"
    reporter.generate_html(results, html_file)
    print(f"\n📄 Reporte HTML generado: {html_file}")
    
    # JSON Export
    json_file = f"{REPORT_DIR}/data_{timestamp}.json"
    reporter.generate_json(results, json_file)
    
    # Cleanup viejo
    backup_mgr.cleanup_old_backups(days=30)
    
    print("\n✅ Auditoría completada exitosamente")

def process_results(raw_data, backup_mgr, analyzer):
    """Procesa datos crudos y enriquece el resultado"""
    result = {
        "hostname": raw_data['hostname'],
        "ip": raw_data['host'],
        "status": "online" if not raw_data.get('errors') else "error",
        "issues": raw_data.get('errors', [])
    }
    
    # Backup
    if raw_data.get('config'):
        meta = backup_mgr.save_config(raw_data['hostname'], raw_data['config'])
        result['backup_taken'] = True
        result['backup_file'] = meta['file']
    
    # Análisis de interfaces
    if 'show ip interface brief' in raw_data.get('commands', {}):
        intf_data = analyzer.analyze_interfaces(
            raw_data['commands']['show ip interface brief']
        )
        result['interfaces_total'] = intf_data.get('total', 0)
        result['interfaces_up'] = intf_data.get('up_up', 0)
        result['issues'].extend(intf_data.get('issues', []))
    
    # Compliance
    if raw_data.get('config'):
        compliance = analyzer.check_compliance(raw_data['config'])
        result['compliance_score'] = compliance['score']
        result['compliance_details'] = compliance['checks']
    
    return result

if __name__ == "__main__":
    main()

#!/usr/bin/env python3
"""
Network Manager Complete - Proyecto Final CCNA Academy
Integra: Netmiko + NAPALM + TextFSM + Jinja2
──────────────────────────────────────────────────────
Uso: python network_manager_complete.py
Requiere: pip install netmiko ntc-templates jinja2 pyyaml
"""

import yaml
import json
import logging
from datetime import datetime
from pathlib import Path
from concurrent.futures import ThreadPoolExecutor, as_completed

from netmiko import ConnectHandler
from ntc_templates.parse import parse_output
from jinja2 import Template

# Configuración
logging.basicConfig(level=logging.INFO, format='%(levelname)s: %(message)s')
logger = logging.getLogger(__name__)

INVENTORY = [
    {"hostname": "R1", "host": "192.168.1.1", "device_type": "cisco_ios", 
     "username": "admin", "password": "cisco123"},
    {"hostname": "SW1", "host": "192.168.1.10", "device_type": "cisco_ios",
     "username": "admin", "password": "cisco123"},
]

COMMANDS = ["show ip interface brief", "show version", "show running-config"]

HTML_TEMPLATE = """



    Network Report {{date}}
    


    

        
Network Infrastructure Report

        
Generated: {{date}}

    

    

        

            
{{stats.total}}

            
Total Devices

        

        

            
{{stats.online}}

            
Online

        

        

            
{{stats.avg_compliance}}%

            
Avg Compliance

        

    

    
        {% for d in devices %}
        
        {% endfor %}
    

        
HostnameIPStatusComplianceBackup

            {{d.hostname}}
            {{d.ip}}
            {{d.status}}
            {{d.compliance}}%
            {{'✓' if d.backup else '✗'}}
        



"""

class NetworkManager:
    def __init__(self):
        self.results = []
        self.backup_dir = Path("backups") / datetime.now().strftime("%Y%m%d")
        self.backup_dir.mkdir(parents=True, exist_ok=True)
    
    def audit_device(self, device):
        """Audita un dispositivo individual"""
        result = {
            "hostname": device['hostname'],
            "ip": device['host'],
            "status": "error",
            "compliance": 0,
            "backup": False,
            "issues": []
        }
        
        try:
            logger.info(f"Conectando a {device['hostname']}...")
            
            with ConnectHandler(**{k: v for k, v in device.items() 
                                  if k in ['device_type', 'host', 'username', 'password']}) as conn:
                
                conn.enable()
                result["status"] = "online"
                
                # Backup
                config = conn.send_command("show running-config")
                backup_file = self.backup_dir / f"{device['hostname']}.cfg"
                with open(backup_file, 'w') as f:
                    f.write(config)
                result["backup"] = True
                
                # Compliance checks
                checks = [
                    "ip ssh version 2" in config,
                    "service password-encryption" in config,
                    "ntp server" in config,
                    "no ip http server" in config
                ]
                result["compliance"] = int(sum(checks)/len(checks)*100)
                
                # Parse interfaces
                output = conn.send_command("show ip interface brief", use_textfsm=True)
                if isinstance(output, list):
                    up = sum(1 for i in output if i.get('status')=='up')
                    result["interfaces_up"] = up
                
        except Exception as e:
            logger.error(f"Error en {device['hostname']}: {e}")
            result["issues"].append(str(e))
        
        return result
    
    def run_audit(self, max_workers=3):
        """Ejecuta auditoría en paralelo"""
        with ThreadPoolExecutor(max_workers=max_workers) as executor:
            futures = {executor.submit(self.audit_device, d): d for d in INVENTORY}
            for future in as_completed(futures):
                self.results.append(future.result())
    
    def generate_report(self):
        """Genera reporte HTML"""
        stats = {
            "total": len(self.results),
            "online": sum(1 for r in self.results if r['status']=='online'),
            "avg_compliance": int(sum(r['compliance'] for r in self.results)/len(self.results))
        }
        
        template = Template(HTML_TEMPLATE)
        html = template.render(
            date=datetime.now().strftime("%Y-%m-%d %H:%M"),
            devices=self.results,
            stats=stats
        )
        
        report_file = f"report_{datetime.now().strftime('%Y%m%d_%H%M')}.html"
        with open(report_file, 'w') as f:
            f.write(html)
        
        print(f"\n✅ Reporte generado: {report_file}")
        print(f"📁 Backups guardados en: {self.backup_dir}")

if __name__ == "__main__":
    manager = NetworkManager()
    manager.run_audit()
    manager.generate_report()
